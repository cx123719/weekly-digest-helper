        (function() {

            // ---------- 数据存储 ----------

            const STORAGE_KEY = 'weekly_records';

            const SETTINGS_KEY = 'weekly_ai_settings';

 

            function loadRecords() {

                const raw = localStorage.getItem(STORAGE_KEY);

                if (!raw) return [];

                try { return JSON.parse(raw); } catch (e) { return []; }

            }

 

            function saveRecords(records) {

                localStorage.setItem(STORAGE_KEY, JSON.stringify(records));

            }

 

            function addRecord(text) {

                const records = loadRecords();

                records.push({

                    id: Date.now().toString(36) + Math.random().toString(36).slice(2, 6),

                    text: text.trim(),

                    time: new Date().toISOString()

                });

                records.sort((a, b) => new Date(b.time) - new Date(a.time));

                saveRecords(records);

                renderRecordList();

            }

 

            function deleteRecord(id) {

                let records = loadRecords();

                records = records.filter(r => r.id !== id);

                saveRecords(records);

                renderRecordList();

            }

 

            function clearAllRecords() {

                if (confirm('确定要清空全部记录吗？此操作不可恢复。')) {

                    saveRecords([]);

                    renderRecordList();

                }

            }

 

            // ---------- 渲染记录列表 ----------

            function renderRecordList() {

                const records = loadRecords();

                const listEl = document.getElementById('recordList');

                if (records.length === 0) {

                    listEl.innerHTML = '<li class="empty-message">暂无记录，先添加一条吧</li>';

                    return;

                }

                listEl.innerHTML = records.map(r => {

                    const date = new Date(r.time);

                    const timeStr = `${date.getMonth()+1}/${date.getDate()} ${date.getHours().toString().padStart(2,'0')}:${date.getMinutes().toString().padStart(2,'0')}`;

                    return `

                        <li class="record-item">

                            <span class="record-time">${timeStr}</span>

                            <span class="record-text">${escapeHtml(r.text)}</span>

                            <button class="delete-btn" data-id="${r.id}" title="删除">✕</button>

                        </li>

                    `;

                }).join('');

                document.querySelectorAll('.delete-btn').forEach(btn => {

                    btn.addEventListener('click', function() {

                        deleteRecord(this.getAttribute('data-id'));

                    });

                });

            }

 

            function escapeHtml(text) {

                const map = { '&': '&', '<': '<', '>': '>', '"': '"', "'": ''' };

                return text.replace(/[&<>"']/g, m => map[m]);

            }

 

            // ---------- 语音识别 ----------

            const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;

            let recognition = null;

            let isRecording = false;

 

            if (SpeechRecognition) {

                recognition = new SpeechRecognition();

                recognition.lang = 'zh-CN';

                recognition.interimResults = false;

                recognition.continuous = false;

 

                recognition.onresult = function(event) {

                    const transcript = event.results[0][0].transcript;

                    if (transcript.trim()) {

                        addRecord(transcript.trim());

                    }

                    stopRecordingUI();

                };

 

                recognition.onerror = function(event) {

                    stopRecordingUI();

                    if (event.error === 'not-allowed') {

                        alert('麦克风权限被拒绝，请在浏览器设置中允许访问麦克风。');

                    } else if (event.error === 'no-speech') {

                        alert('没有检测到语音，请重试。');

                    } else {

                        alert('语音识别出错: ' + event.error);

                    }

                };

 

                recognition.onend = function() {

                    stopRecordingUI();

                };

            }

 

            function startRecordingUI() {

                isRecording = true;

                const micBtn = document.getElementById('micBtn');

                micBtn.classList.add('recording');

                micBtn.textContent = '��';

                document.getElementById('textInput').placeholder = '正在聆听...';

            }

 

            function stopRecordingUI() {

                isRecording = false;

                const micBtn = document.getElementById('micBtn');

                micBtn.classList.remove('recording');

                micBtn.textContent = '��';

            }

 

            function toggleMic() {

                if (!SpeechRecognition) {

                    alert('当前浏览器不支持语音识别，请使用 Chrome 或 Edge。');

                    return;

                }

                if (isRecording) {

                    recognition.stop();

                    stopRecordingUI();

                } else {

                    startRecordingUI();

                    recognition.start();

            }

 

            // ---------- 生成普通汇报 ----------

            function getFilteredRecords() {

                const startDateVal = document.getElementById('startDate').value;

                const endDateVal = document.getElementById('endDate').value;

                if (!startDateVal || !endDateVal) return null;

 

                const start = new Date(startDateVal);

                start.setHours(0,0,0,0);

                const end = new Date(endDateVal);

                end.setHours(23,59,59,999);

 

                if (start > end) return null;

 

                let records = loadRecords();

                return records.filter(r => {

                    const t = new Date(r.time);

                    return t >= start && t <= end;

                }).sort((a, b) => new Date(a.time) - new Date(b.time));

            }

 

            function generateReportText(records) {

                if (!records || records.length === 0) return '所选日期范围内没有记录。';

 

                let reportLines = [];

                let currentDateStr = '';

                let counter = 1;

 

                records.forEach(r => {

                    const d = new Date(r.time);

                    const dateStr = `${d.getFullYear()}-${(d.getMonth()+1).toString().padStart(2,'0')}-${d.getDate().toString().padStart(2,'0')}`;

                    if (dateStr !== currentDateStr) {

                        currentDateStr = dateStr;

                        reportLines.push(`\n【${currentDateStr}】`);

                        counter = 1;

                    }

                    reportLines.push(`${counter}. ${r.text}`);

                    counter++;

                });

 

                return reportLines.join('\n').trim();

            }

 

            function generateNormalReport() {

                const records = getFilteredRecords();

                if (!records) {

                    reportArea.textContent = '请选择正确的日期范围。';

                    return;

                }

                reportArea.textContent = generateReportText(records);

            }

            function loadSettings() {

            function saveSettings(settings) {

            // ---------- AI 提炼汇报 ----------

            async function generateAIReport() {

                const records = getFilteredRecords();

                const reportArea = document.getElementById('reportArea');

                if (!records) {

                    reportArea.textContent = '请选择正确的日期范围。';

                    return;

                }

                if (records.length === 0) {

                    reportArea.textContent = '所选日期范围内没有记录。';

                    return;

                }

 

                const settings = loadSettings();

                if (!settings.apiUrl || !settings.apiKey) {

                    reportArea.textContent = '请先配置 AI 设置（API 地址和 Key）。';

                    document.getElementById('settingsPanel').classList.add('active');

                    applySettingsToForm();

                    return;

                }

 

                reportArea.textContent = 'AI 正在提炼中，请稍候...';

 

                // 构建提示词：把所有原始记录按时间顺序列出

                const rawList = records.map(r => {

                    const d = new Date(r.time);

                    const dateStr = `${d.getFullYear()}-${(d.getMonth()+1).toString().padStart(2,'0')}-${d.getDate().toString().padStart(2,'0')}`;

                    return `[${dateStr}] ${r.text}`;

                }).join('\n');

 

                const prompt = `你是一个专业的工作汇报助手。请将以下每日琐碎工作记录，整理成正式、简洁的条目式周报。要求：

1. 按日期分组，日期格式为【YYYY-MM-DD】。

2. 每个日期下，提炼出2-5条核心工作，去掉口语化表达，使用正式、结果导向的语言。

3. 每条以序号开头，每条不超过30字。

4. 直接输出汇报内容，不要加任何解释或前缀。

 

原始记录：

${rawList}`;

 

                const messages = [

                    { role: "system", content: "你是一个专业的工作汇报助手。" },

                    { role: "user", content: prompt }

                ];

 

                try {

                    const response = await fetch(settings.apiUrl, {

                        method: 'POST',

                        headers: {

                            'Content-Type': 'application/json',

                            'Authorization': `Bearer ${settings.apiKey}`

                        },

                        body: JSON.stringify({

                            model: settings.model,

                            messages: messages,

                            temperature: 0.7,

                            max_tokens: 1500

                        })

                    });

 

                    if (!response.ok) {

                        const errText = await response.text();

                        throw new Error(`API 请求失败 (${response.status}): ${errText}`);

                    }

 

                    const data = await response.json();

                    const content = data.choices?.[0]?.message?.content?.trim();

                    if (content) {

                        reportArea.textContent = content;

                    } else {

                        throw new Error('AI 返回内容为空');

                    }

                } catch (error) {

                    console.error(error);

                    reportArea.textContent = `AI 提炼失败: ${error.message}`;

                }

            }

            function copyReport() {

                const reportArea = document.getElementById('reportArea');

                const text = reportArea.textContent;

                if (!text || text.includes('请选择') || text.includes('没有记录') || text.includes('正在提炼')) {

                    alert('没有可复制的汇报内容');

                    return;

                }

                navigator.clipboard.writeText(text).then(() => {

                    alert('汇报内容已复制到剪贴板 ✅');

                }).catch(() => {

                    alert('复制失败，请手动复制');

                });

            }
